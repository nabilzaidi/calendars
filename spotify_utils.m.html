class Spotify:
    """
    Classe représentant le client Spotify.\n
    Facilite la connexion, le rafraichissement des token
    et la recherche de musique sur Spotify.
    """

    def __init__(self, scope='user-library-read',
                 username="nabilzaidi",
                 client_id="a469c28537484cdbb4bae55a7fae6fe4",
                 client_secret="35d08f7f75294e6d921b14a33d2299f0",
                 redirect_uri="http://localhost/"):
        """
        Fonction d'initialisation. Les informations sont disponibles
        sur l'interface application de Spotify.\n
        @var scope [string], ce que le client est authorisé à faire,
        paramètre fournie et modifiable dans l'application Spotify.
        @var username [string], nom d'utilisateur Spotify\n
        @var client_id [string], id du client Spotify\n
        @var client_secret [string], clé secrète du client Spotify\n
        @var redirect_uri [string], url de redirection choisie sur Spotify.\n
        """

        self.sp_oauth = spotipy.oauth2.SpotifyOAuth(
            client_id=client_id,
            client_secret=client_secret,
            redirect_uri=redirect_uri,
            scope=scope)
        # Création de l'authentificateur du client Spotify
        self.sp = None
        self.token_info = None

    def init_sp(self):
        """
        Fonction pour initialiser le client Spotify.
        """
        self.token_info = self.sp_oauth.get_cached_token()
        if not self.token_info:
            # Si aucun token n'est dans le cache
            # on récupère l'url qui va nous fournir le code
            # d'authentification
            auth_url = self.sp_oauth.get_authorize_url()
            print(auth_url)
            response = input(
                'Accéder au lien ci dessus, \
et coller le lien de redirection ci dessous: ')
            # Le lien affiché permet de s'authentifier.
            # Spotify redirige ensuite vers le site spécifier dans
            # redirect_uri. Le lien de redirection contient en réalité
            # le code permettant de s'authentifier.
            code = self.sp_oauth.parse_response_code(response)
            token_info = self.sp_oauth.get_access_token(code)

        token = token_info['access_token']
        # Initialisation du client Spotify avec le token récupéré
        self.sp = spotipy.Spotify(auth=token)

    def refresh(self):
        """
        Fonction pour rafraichir le token Spotify
        """
        self.token_info = self.sp_oauth.refresh_access_token(
            self.token_info['refresh_token']
        )
        token = self.token_info['access_token']
        self.sp = spotipy.Spotify(auth=token)

    def get_song(self, song, n_try=2, limit=20):
        """
        Fonction pour trouver une musique sur Spotify.\n
        Cette fonction va "nettoyer" le titre pour simplifier la recherche.
        @var song [string], la musique à chercher\n
        @var n_try [int], le nombre d'essai à effectuer\n
        @var limit [int], le nombre maximum de résultats à récupèrer\n
\n
        @returns [dict], un dictionnaire avec:\n
            "query": la requête,\n
            "res": les résultats Spotify\n
            "n_items": le nombre de titres trouvés par Spotify
        """
        aux_search = lambda s: self.sp.search(q=s, limit=limit, market="FR")
        len_fct = lambda res: res["res"].get("tracks", dict()).get("total", 0)
        return get_song(search_fct=aux_search,
                        song=song, n_try=2, refresh=lambda: True,
                        length_res_fct=len_fct)

def item_simplifie(item):
    """
    Renvoie une version simplifié d'une musique trouvée sur Spotify.\n
    @var item [dict], musique trouvée sur Spotify
    (avec l'album, les différents artistes...)
    \n
    @returns [dict], un dictionnaire avec
        "id": id Spotify de la musique,\n
        "song_name": nom de la musique,\n
        "album_name": nom de l'album,\n
        'album_release_date': date de sortie de l'album,\n
        "artists": nom des artistes
    """
    return {
        "id": item["id"],
        "song_name": item["name"],
        "album_name": item["album"]["name"],
        'album_release_date': item["album"]['release_date'],
        "artists": ", ".join([a["name"] for a in item["artists"]])
    }


def res_simplifie(res):
    """
    Renvoie une version simplifié d'un résultat Spotify.\n
    @var res [dict], résultat trouvé sur Spotify
    (ensemble de musiques pouvant correspondre à la requête)
    \n
    @returns [dict], un dictionnaire avec
        "items": liste des résultats simplifiés
        (retournés par item_simplifie),\n
        "n_items": nombre de résultats trouvés,\n
    """
    items = res.get("tracks", dict()).get("items", [])
    new_res = dict()
    l_items_simplifies = []
    for i in items:
        if i is not None:
            info = item_simplifie(i)
            l_items_simplifies.append(info)

    new_res["items"] = l_items_simplifies
    new_res["n_items"] = len(l_items_simplifies)

    return new_res

def compare_res_base(res, title, artist):
    """
    Compare une musique trouvée sur Spotify
    avec le titre et l'artiste de la base de donnée.
    @var res [dict], musique trouvée sur Spotify
    @var title [string], titre dans la base de donnée
    @var artist [string], atiste dans la base de donnée
    \n
    @returns [dict], dictionnaire avec les scores:
        "song": scores appliqués sur le titre de la musique trouvée
        et le titre de la base de données.
        "album": scores appliqués sur l'album de la musique trouvée
        et le titre de la base de données.
        "artist": scores appliqués sur l'artiste de la musique trouvée
        et l'artiste dans la base de données.
    """
    scores_song = all_scores(res["song_name"], title)
    scores_album = all_scores(res["album_name"], title)
    scores_artist = all_scores(res["artists"], artist)
    return {"song": scores_song,
            "album": scores_album,
            "artist": scores_artist
            }
