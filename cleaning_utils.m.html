tools.cleaning_utils module

Ensemble d'outils pour nettoyer les titres et les artistes.

Les fonctions commençant par "clean_" seront utilisées pour nettoyer les titres alors que les fonctions commençant par "remove_" seront utilisées lors de la recherche de musique pour simplifier la recherche.

#!/usr/bin/env python
# coding: utf-8

"""
Ensemble d'outils pour nettoyer les titres et les artistes.\n
Les fonctions commençant par "clean_" seront utilisées pour nettoyer
les titres alors que les fonctions commençant par "remove_" seront
utilisées lors de la recherche de musique pour simplifier la recherche.
"""


import re
from unidecode import unidecode

string_punctuation = '"#$%&\'()*+-/:;<=>?@[\\]^_`{|}~'
"""
<string>, Ensemble des ponctuations à supprimer
"""


def clean_casse(nom):
    """ Corrige la casse\n
    Céline Dion -> céline dion\n
    @var nom <string>, chaine de caractère à nettoyer\n
    \n
    @returns <string> chaine de caractère nettoyée.
    """
    nom = nom.lower()
    return nom


def clean_feat(nom):
    """ Mets les collaborations sous forme :\n
     Artiste 1, Artiste 2, ..., Artiste n\n
    Céline Dion feat Johnny Hallyday -> Céline Dion, Johnny Hallyday\n
    @var nom <string>, chaine de caractère à nettoyer\n
\n
    @returns <string> chaine de caractère nettoyée\n
    """
    collaborations = ['feat. ', 'ft. ', '&', ';',
                      "and ", "/", "vs", "et ", "featuring "]
    big_regex = re.compile('|'.join(map(re.escape, collaborations)))
    nom = big_regex.sub(",", nom)
    return nom


def clean_punctuation(nom):
    """ Corrige signe de ponctuation\n
    (à utiliser après clean_feat)\n
    @var nom <string>, chaine de caractère à nettoyer\n
\n
    @returns <string> chaine de caractère nettoyée\n
    """
    nom = nom.translate(str.maketrans(
        '', '', string_punctuation))  # Ponctuation

    return nom


def clean_accents(nom):
    """ Corrige accents\n
    Céline Dion -> Celine Dion\n
    @var nom <string>, chaine de caractère à nettoyer\n
\n
    @returns <string> chaine de caractère nettoyée\n
    """
    nom = unidecode(nom)
    # nom = unicodedata.normalize('NFKD', nom)\
    #     .encode('ascii', 'ignore')\
    #     .decode('utf-8')
    return nom


def clean_CD(nom):
    """ Mots superflus qui decrivent le CD\n
    @var nom <string>, chaine de caractère à nettoyer\n
\n
    @returns <string> chaine de caractère nettoyée\n
    """
    CD = ['special edition', "cast", "karaoke", "mixed"]
    big_regex = re.compile('|'.join(CD))
    # re.compile('|'.join(map(re.escape, CD)))
    nom = big_regex.sub("", nom)
    return nom


def clean_descriptifs(nom):
    """ Qualificatifs des chanteurs\n
    Baryton - Nom  Artiste pas ok\n
    Barytones de Sevilla ok\n
    @var nom <string>, chaine de caractère à nettoyer\n

    @returns <string> chaine de caractère nettoyée\n
    """
    descriptifs = [r'\bbarython\b', r'\bsoprano\b', r'\btenor\b']
    big_regex = re.compile('|'.join(descriptifs))
    nom = big_regex.sub("", nom)
    return nom


def clean_spaces(nom):
    """ Nettoie les espaces\n
    Après les clean_ , avant clean_order\n
    "Céline    Dion" -> "Céline Dion"\n
    @var nom <string>, chaine de caractère à nettoyer\n
\n
    @returns <string> chaine de caractère nettoyée\n
    """
    virgule = re.compile(r"\s*,\s*")
    nom = re.sub(virgule, ", ", nom).replace("  ", " ")
    return nom


def clean_order(nom):
    """ Classement alphabétique intra et interartistes\n
    Dion Céline, Johnny Hallyday -> Céline Dion, Hallyday Johnny\n
    @var nom <string>, chaine de caractère à nettoyer\n
\n
    @returns <string> chaine de caractère nettoyée\n
    """
    l_nom = nom.split(", ")
    cleaned_l_nom = []
    for n in l_nom:
        l_n = n.split()
        l_n.sort()
        cleaned_l_nom.append(" ".join(l_n))
    cleaned_l_nom.sort()
    return ", ".join(cleaned_l_nom)


func_clean = [
    clean_casse,
    clean_feat,
    clean_punctuation,
    clean_accents,
    clean_CD,
    clean_descriptifs,
    clean_spaces,
    clean_order
]
"""
[list], Liste des fonctions de nettoyage
"""


def clean_full(nom):
    """ Fonction de nettoyage finale.\n
    Itère sur func_clean pour nettoyer une chaine de caractère.\n
    @var nom <string>, chaine de caractère à nettoyer\n
\n
    @returns <string> chaine de caractère nettoyée\n
    """

    cleaned_name = nom

    if isinstance(nom, str):
        for func in func_clean:
            cleaned_name = func(cleaned_name)

    return cleaned_name


def clean_song(song):
    """
    Fonction auxiliaire. Elle peut s'appliquer\n
    à un couple de chaine de caractère (titre, artiste)\n
    ou à une chaine de caractère.\n
    @var nom <string / tuple>, chaine de caractère à nettoyer\n
\n
    @returns <string> chaine de caractère nettoyée
    """
    cleaned = song
    if isinstance(song, str):
        cleaned = clean_full(cleaned)
    else:
        cleaned = (clean_full(song[0]), clean_full(song[1]))
    return cleaned


def remove_brackets(song):
    """
    Enlèves les parenthèses et les crochets.\n
    @var nom <string>, chaine de caractère à nettoyer\n
\n
    @returns <string> chaine de caractère nettoyée\n
    """
    regex = r"\(.*\)|\[.*\]"
    regex2 = r"\(.*"
    regex3 = r"\[.*"
    query = re.sub(regex, "", song)
    if "(" in query:
        query = re.sub(regex2, "", query)
    if "[" in query:
        query = re.sub(regex3, "", query)
    query = query.strip()
    return query


def remove_album(song):
    """
    Supprime tout ce qu'il y a après un tiret.
    Souvent dans les titres, ce qui suit un tiret
    est le nom de l'album.\n
    @var nom <string>, chaine de caractère à nettoyer\n
\n
    @returns <string> chaine de caractère nettoyée\n
    """
    regex = r"\s-\s.*"
    query = re.sub(regex, "", song)
    query = query.strip()
    return query


def remove_details(song):
    """
    Supprime tout ce qu'il y a après un deux-point.
    Souvent dans les titres, ce qui suit un deux-point
    correspond à des détails inutiles.\n
    @var nom <string>, chaine de caractère à nettoyer\n
\n
    @returns <string> chaine de caractère nettoyée\n
    """
    regex = r":[^:]*$"
    song = re.sub(regex, "", song)
    return song.strip()
